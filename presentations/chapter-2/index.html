Pipe purity


@Pipe({
  name: 'timer',
  pure: true, // default: true
})
export class TimerPipe implements PipeTransform {
  transform(time: { minutes: number; seconds: number }) {
    return `${time.minutes} : ${time.seconds}`;
  }
}
