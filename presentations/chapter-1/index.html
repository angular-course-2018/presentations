fancy-button.component.ts

@Component({/.../})
export class FancyButtonComponent {
  @Output() pick: EventEmitter<string> = new EventEmitter();
}

fancy-button.component.html

parent.component.html

<app-fancy-button (pick)="window.alert($event)"/>

Event binding is done using round brackets ().

Testing a Component

1. Describe what you're going to test
	
describe('FancyButtonComponent', () => {

  it('should display initial value as button label');
  it('should change button label when changing input value');
  it('should emit `pressed` event when clicking the button');
  it('should display default button label when input value is not defined');

});
2. Configure testing module

import { TestBed, async } from '@angular/core/testing';
import { AppComponent } from './app.component';
import { NO_ERRORS_SCHEMA } from '@angular/core';

describe('AppComponent', () => {

  beforeEach(async(() => {
    TestBed.configureTestingModule({
      declarations: [
        AppComponent
      ],
      schemas: [ NO_ERRORS_SCHEMA ]
    }).compileComponents();
  }));
});
3. Create component instance and use it

import { TestBed, async } from '@angular/core/testing';
import { AppComponent } from './app.component';
	
describe('AppComponent', () => {
  beforeEach(async(() => { /.../ }));

  it('should create the app', () => {
    const fixture = TestBed.createComponent(AppComponent);
    const app = fixture.debugElement.componentInstance;
	
    expect(app).toBeTruthy();
  });

});
