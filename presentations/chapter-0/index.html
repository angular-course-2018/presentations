A simple request JS way

					
// returns response with data property and transactions array
async function getAccountData(request){
  const response = await fetch('http://my-bank/data', request);
  return response.json();
}

// id not _id 
const accountData = await getAccountData({ id: 12345 });

// firstTransaction of type Transaction
const firstTransaction = accountData.data.transactions[0];

Same request TS way

						
async function getAccountData(request: AccountRequest): AccountResponse {
  const response = await fetch('http://my-bank/data', request);
  return response.json();
}

const request = new AccountRequest(12345);
const accountData = await getAccountData(request);

const firstTransaction = accountData.data.transactions[0];

Spec file structure


describe('test unit', () => {
  beforeAll(() => {
    // once before the unit
  });
  beforeEach(() => {
    // once before every test
  });
  afterEach(() => {
    // once after every test
  });
  afterAll(() => {
    // once after the unit
  });
});

Given, When, Then

Explicit is better than implicit.
PEP20 - Python Zen Code

it('should calculate 115USD for 100USD money and 15% tax', () => {
  // given
  expect.hasAssertions();
  const calculator = new SimpleTaxCalculator();

  // when
  const result = calculator.calc(100, 15);

  // then
  expect(result).toBe(115);
});

Before Each


describe('simple calculator', () => {
  let calculator: SimpleTaxCalculator;
  beforeEach(() => {
    calculator = new SimpleTaxCalculator();
  });
  
  it('should calculate 115USD for 100USD money and 15% tax', () => {
    // given
    expect.hasAssertions();
    // when
    const result = calculator.calc(100, 15);
    // then
    expect(result).toBe(115);
  });
});
