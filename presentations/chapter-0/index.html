  const response = await fetch('http://localhost/data', request);
  return response.json();
}

// id not _id

export class GroupOfPeople<T extends Person> {
  constructor(public group: T[]) {}
  makeThemTalk() {
    for (const person of this.group) {
      person.talk();
    }
  }
}
